18.四数之和

Apr 3, 2020 #题解 #Leetcode

题意：找四个数看能否等于target

分析：在三数之和基础上，外面再套一层循环，给出去重判断条件，有相同的就继续，这里注意第一个相同的是要做的，只有第二个相同的才继续往下跳，否则【2，2，2，2】8 这种就会返回空。

AC：

class Solution {
public:
    vector<vector<int>> fourSum(vector<int>& nums, int target) {
        vector<vector<int>> res;
        if(nums.size()<4)
            return res;
        
        sort(nums.begin(),nums.end());
        for(int z=0;z<=nums.size()-4;z++)
        {
            if(z>0&&nums[z]==nums[z-1])
                    continue;
            int tar=target-nums[z];  //转为3数之和
            for(int i=z+1;i<=nums.size()-3;i++)  
            {
                
                if(i>z+1&&nums[i]==nums[i-1])
                    continue;
                int t=tar-nums[i];  //转为两数之和

                //cout<<"target"<<nums[i]<<endl;;
                int l=i+1,r=nums.size()-1;
                while(l<r)  
                {
                    if(nums[l]+nums[r]-t>0)
                        r--;
                    else if(nums[l]+nums[r]-t<0)
                        l++;
                    else
                    {
                        res.push_back({nums[z],nums[i],nums[l],nums[r]});
                        while (l < r && nums[l] == nums[++l]);
                        while (l < r && nums[r] == nums[--r]);
                    }           
                }
            }
        }
        
        return res;
    }
};
