43. 字符串相乘

Apr 3, 2020 #题解 #Leetcode #大数运算 #字符串

题意：大数相乘，给的字符串

AC：

class Solution {
public:
    string multiply(string num1, string num2) {
        //大数乘法
        //每个数相乘得到的结果都是存放在i+j的位置上i和j都从0开始
        int n1=num1.size();
        int n2=num2.size();
        vector<int> res(n1+n2+1,0);  //存最后结果
        
        //设定一个进位数组，逆向保存在数组里
        int maxpos=0;
        for(int i=n1-1;i>=0;i--)
        {
            int carry=0;
            int pos=0;
            vector<int> tmp(n1+n2+1,0); //存计算的每一行结果
            for(int j=n2-1,k=0;j>=0;j--,k++)   //这层循环完一遍就会产生一行结果，然后加上carry，赋给res
            {
                int a=num1[i]-'0';
                int b=num2[j]-'0';
                int t=a*b;
                //cout<<t<<" 两个数相乘"<<endl;
                //相乘一定不超过两位数，把十位赋给carry的对应位置，个位给tmp的对应位置
                tmp[k-i+n1-1]=(t+carry)%10;   //这个位置是反的
                //cout<<tmp[k-i+n1-1]<<"  本位是"<<"存在："<<k-i+n1-1<<endl; 
                
                carry=(t+carry)/10; 
                //cout<<carry<<"  进位是"<<endl;
                pos=k-i+n1-1; //记录最后的位置
            }
            if(carry!=0)
            {
                tmp[pos+1]=carry;
                //cout<<"有剩下的"<<pos+1 <<endl;
            }
                
            int car=0;
            for(int i=0;i<n1+n2+1;i++)
            {   //cout<<"结果是："<<res[i]<<"+"<<tmp[i]<<"+"<<car<<endl;
                int t=res[i]+tmp[i]+car;
                res[i]=(res[i]+tmp[i]+car)%10;
                //cout<<"=="<<res[i]<<endl;
                car=t/10;
                if(res[i]!=0&&car==0)
                    maxpos=i;
            }
        }
        //reverse(res.begin(),res.end()); 
        //测试输出
        
        string ress="";
        for(int i=maxpos;i>=0;i--)
        {
            if(res[i]>9)
            {
                ress+=(res[i]/10)+'0';
                ress+=res[i]%10+'0';
            }
            else
            {
                char tt=res[i]+'0';
                ress+=tt;
            }
        }
        return ress;
    }
};
