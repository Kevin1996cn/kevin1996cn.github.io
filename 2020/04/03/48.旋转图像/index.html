48.旋转图像

Apr 3, 2020 #题解 #Leetcode

题意：给你一个二维数组，给他顺时针旋转90度。

分析：给两重循环，外层控制进入第几层。

​ 内层控制向左向右向上向下移动的位置。

AC:

class Solution {
public:
    void rotate(vector<vector<int>>& matrix) {
        int n=matrix.size()-1; //n=维数-1
        int layers;
        if((n+1)%2==0)
            layers=(n+1)/2;
        else
            layers=n/2;
        cout<<"cengshu:"<<layers<<endl;
        for(int start=0;start<layers;start++)
        {
            for(int i=0;i<=n-1-start*2;i++)  //
            {
                
                int tmp=matrix[n-start-i][start];
                //cout<<matrix[n-start-i][start]<<"="<<matrix[n-start][n-start-i]<<endl;
                matrix[n-start-i][start]=matrix[n-start][n-start-i];
                //cout<<matrix[n-start][n-start-i]<<"="<<matrix[start+i][n-start]<<endl;
                matrix[n-start][n-start-i]=matrix[start+i][n-start];
                //cout<<matrix[start+i][n-start]<<"="<<matrix[start][start+i]<<endl;
                matrix[start+i][n-start]=matrix[start][start+i];
                //cout<<matrix[start][start+i]<<"="<<tmp<<endl;
                matrix[start][start+i]=tmp;
            }
        }
        
    }
};
