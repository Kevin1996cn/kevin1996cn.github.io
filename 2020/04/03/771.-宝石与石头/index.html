771. 宝石与石头

Apr 3, 2020 #题解 #Leetcode #字符串

题意：J中a和A代表宝石类型，找出S中有多少宝石。

示例 1:

输入: J = "aA", S = "aAAbbbb"
输出: 3

示例 2:

输入: J = "z", S = "ZZ"
输出: 0

AC：

class Solution {
public:
int numJewelsInStones(string J, string S) {
        int a='a',z='z',A='A',Z='Z';
        int daxie[26]={0};
        int xiaoxie[26]={0};
        int res=0;
        for(int i=0;i<J.size();i++)
        {
            int t=J[i];
            if(t>='a'&&t<='z')
            {
                xiaoxie[t-'a']++;
            }
            else
            {
                daxie[t-'A']++;
            }
        }
        for(int j=0;j<S.size();j++)
        {
            int t=S[j];
            if(t>='a'&&t<='z')
            {
                if(xiaoxie[t-'a'])
                    res++;
            }
            else
            {
                if(daxie[t-'A'])
                    res++;
            }
        }
        return res;
    }
};
