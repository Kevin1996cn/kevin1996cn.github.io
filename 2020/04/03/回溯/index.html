22. 括号生成

给你一个数字n，让你求出所有有效的括号组合。

分析：

把这个问题想象成一个树，每个节点可以走左括号或者右括号。

要注意写递归的时候，临时传参不能用引用类型，这样你就会进入每次出来还要减去之前的那个误区。

AC：

class Solution {
public:
    void digui(string res,int left,int right,int sum,vector<string>& ans)
    {
        
        if(left>sum/2||right>sum/2||right>left)    //这个过滤条件就完全足够了吗？
            return;
        if(res.size()==sum)
            ans.push_back(res);
        digui(res+'(',left+1,right,sum,ans);
        digui(res+')',left,right+1,sum,ans);   
        //这边res+‘（’是暂时加上去的 ，函数的参数string res不能传引用，传引用就不是暂时加上去了。           //包括left right都是这样
        
    }
    
    vector<string> generateParenthesis(int n) {
        string res="(";
        vector<string> ans;
        digui(res,1,0,2*n,ans);
        //思路，利用递归的方式，把所有可能的组合写出来，其中有些肯定不会有效的情况不生成。
        return ans;
    }
};

39. 组合总和

给你一个数组，再给你一个target，让你从数组中取数加和=target

思路：

回溯即可。

但是要注意这个不能有重复的组合。比如2、2、3和2、3、2这种不允许出现。

只要给一个接下去走遍历的时候只能走这个i后面的，比如2、3、4、5，你先拿了2，3，再往下走只能走3和它后面的。

AC:

class Solution {
public:
    void fun(vector<int> candidates,int target,int sum,vector<vector<int>>& res,vector<int>& tmp,int i)
    {
        //一个n叉树的dfs
        if(sum>target)
        {
            return;
        }
        if(sum==target)
        {
            res.push_back(tmp);
        }
        for(;i<candidates.size();i++)
        {
            tmp.push_back(candidates[i]);
            fun(candidates,target,sum+candidates[i],res,tmp,i);
            tmp.pop_back();
        }
    }
        
    vector<vector<int>> combinationSum(vector<int>& candidates, int target) {
        vector<vector<int>> res;
        vector<int> tmp;
        int sum=0;
        int i=0;
        fun(candidates,target,sum,res,tmp,i);
        return res;
    }
};

40. 组合总和 II

题意：

和上边一样，只不过这次一个数只能用一次了

分析：

我们只要下一次让i+1，然后用个set去重就行了。 这题不知道为啥不能用unordered_set用set就可以。

AC：

class Solution {
public:
    void fun(vector<int> candidates,int target,int sum,set<vector<int>>& s,vector<int>& tmp,int i)
    {
        //一个n叉树的dfs
        if(sum>target)
        {
            return;
        }
        if(sum==target)
        {
            vector<int> tt=tmp;
            sort(tt.begin(),tt.end());
            s.insert(tt);
            
            //res.push_back(tmp);
        }
        for(;i<candidates.size();i++)
        {
            tmp.push_back(candidates[i]);
            fun(candidates,target,sum+candidates[i],s,tmp,i+1);
            tmp.pop_back();
        }
    }
        
    vector<vector<int>> combinationSum2(vector<int>& candidates, int target) {
        vector<vector<int>> res;
        vector<int> tmp;
        set<vector<int>> s;
        int sum=0;
        int i=0;
        fun(candidates,target,sum,s,tmp,i);
        
        for(auto vec:s)
        {
            res.push_back(vec);
        }
        return res;
    }  
};
