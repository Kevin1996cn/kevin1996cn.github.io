392.判断子序列

题意：判断S是否是T的字序列，可以不挨着。

例：s=abc ，t=axbdb 返回true

AC:

O(n)复杂度

class Solution {
public:
    bool isSubsequence(string s, string t) {
        long long int slen=s.size();
        long long int tlen=t.size();
        if(slen>tlen) return false;
        if(slen==0) return true;
        int j=0;
        for(int i=0;i<=tlen;i++)
        {
            if(s[j]==t[i])
                j++;
            if(j==slen)
                return true;
        }
     return false;   
    }
};

dp方案只能是在t长度不大的时候可行，因此方法仅是一个可行方案。

class Solution {
public:
    bool isSubsequence(string s, string t) {
        int slen=s.size();
        int tlen=t.size();
        if(slen>tlen)
            return false;
        if(slen==0)
            return true;
        int dp[slen+1][tlen+1];
        memset(dp,0,sizeof(dp));
        for(int j=0;j<=tlen;j++)
        {
            dp[0][j]=true;
        }
        for(int i=1;i<=slen;i++)
        {
            for(int j=1;j<=tlen;j++)
            {
                if(s[i-1]==t[j-1])
                    dp[i][j]=dp[i-1][j-1];
                else
                    dp[i][j]=dp[i][j-1];
            }
        }
     return dp[slen][tlen];   
    }
};

  1. 买卖股票的最佳时机含手续费

题意：给你 每天股票的价格，让你多次买入卖出，买入股票卖出之后才能再次买入，求你能获得的最大利润。交易有手续费fee。

输入: prices = [1, 3, 2, 8, 4, 9], fee = 2
输出: 8
解释: 能够达到的最大利润:  
在此处买入 prices[0] = 1
在此处卖出 prices[3] = 8
在此处买入 prices[4] = 4
在此处卖出 prices[5] = 9
总利润: ((8 - 1) - 2) + ((9 - 4) - 2) = 8.

AC：

class Solution {
public:
    int maxProfit(vector<int>& prices, int fee) {
        int n=prices.size();
        int cash=0,hold=-prices[0];
        //cash代表空仓的时候，我们手里的利润
        //hold代表我们买入，手里的利润
        cout<<"cash:"<<cash<<" hold:"<<hold<<endl;
        for(int i=1;i<=n-1;i++)
        {
            cout<<"price:"<<prices[i]<<endl;
            cash=max(cash,hold+prices[i]-fee);
            hold=max(hold,cash-prices[i]);
            cout<<"cash:"<<cash<<" hold:"<<hold<<endl;
        }
        return cash;
    }
};

  1. 最长公共子序列

题意：找两个字符串的最长子序列

分析：dp【i】【j】代表第一个串的从第一个字母到第i个字母的字串与第二个串从一个字母到第j个字母的公共字串长度。

​ dp代表的是前一个的长度，所以最后返回dp【rows】【cols】而不是-1的形式。

AC:

// 1.　S{s1,s2,s3....si} T{t1,t2,t3,t4....tj}

// 2.　子问题划分

// (1) 如果S的最后一位等于T的最后一位，则最大子序列就是{s1,s2,s3...si-1}和{t1,t2,t3...tj-1}的最大子序列+1

// (2) 如果S的最后一位不等于T的最后一位，那么最大子序列就是

// ① {s1,s2,s3..si}和 {t1,t2,t3...tj-1} 最大子序列

// ② {s1,s2,s3...si-1}和{t1,t2,t3....tj} 最大子序列

// 以上两个自序列的最大值

// 3.　边界

// 只剩下{s1}和{t1}，如果相等就返回1，不等就返回0

// 4.　使用一个表格来存储dp的结果

// 如果 S[i] == T[j] 则dp[i][j] = dp[i-1][j-1] + 1

// 否则dp[i][j] = max(dp[i][j-1],dp[i-1][j])

class Solution {
public:
    int max3(int x,int y ,int z){
        return max(x,max(y,z));
    }
    int longestCommonSubsequence(string text1, string text2) {
        int rows=text1.size();
        int cols=text2.size();
        vector<vector<int>> dp(rows+1,vector<int>(cols+1, 0));
        //二维数组
        for(int i=1;i<=rows;i++)
        {
            for(int j=1;j<=cols;j++)
            {

                dp[i][j] = max3(dp[i - 1][j], dp[i][j - 1], 
                                dp[i - 1][j - 1] + (text1[i - 1] == text2[j - 1]));
            }
        }
        return dp[rows][cols];
    }
};

  1. 完全平方数

题意：给一个数，可以由多个完全平方数组成，求最少要几个完全平方数

如5=4+1 就输出2

3=1+1+1 输出3

AC：

class Solution {
public:
    int numSquares(int n) {
        vector<int> dp(n+1,0);
        for(int i=0;i<=n;i++)
        {
            dp[i]=i;
            for(int j=1;i-j*j>=0;j++)
            {
                dp[i]=min(dp[i],dp[i-j*j]+1);   //减去一个完全平方数，然后加一，看哪个小          
            }
        }
        return dp[n];
    }
};

  1. 三角形最小路径和

题意：给你一个三角形，然后找出从一条路径使得最后的加和最短。

[
     [2],
    [3,4],
   [6,5,7],
  [4,1,8,3]
]

分析：从下到上，从倒数第二排开始，这里每一位都应该是，它和它下边两个数字的和的最小值。比如6应该现在等于7，5应该等于6，7应该等于3，这样一层一层上去，triangle【0】【0】就是最后的最短路径的和了。

AC：

class Solution {
public:
    int minimumTotal(vector<vector<int>>& triangle) {
        int n=triangle.size();
        for(int i=n-2;i>=0;i--)
        {
            for(int j=0;j<triangle[i].size();j++)
            {
                triangle[i][j]+=min(triangle[i+1][j],triangle[i+1][j+1]);
            }
        }
        return triangle[0][0];
    }
};
  1. 最长上升子序列

题意：给你一个数组，找到最长的递增序列，数字不重复

AC：

class Solution {
public:
    int lengthOfLIS(vector<int>& nums) {
        int n=nums.size();
        if(n==0)
            return 0;
        int dp[n];
        dp[0]=1;
        int res=1;
        //从第二个数开始，如果这个数比前面的数小，就继续往前走，直到找到比它数字值小的但是dp值最大的那个。
        for(int i=1;i<n;i++)
        {   
            dp[i]=1;
            int max=1;
            for(int j=i-1;j>=0;j--)
            {
                if(nums[i]>nums[j])
                {
                    dp[i]=dp[j]+1;
                    //cout<<"第"<<i<<"个数的值"<<dp[i]<<endl;
                    if(dp[i]>max)
                        max=dp[i];
                }
            }
            dp[i]=max;
            if(dp[i]>res)
                res=max;
        }
        
        for(int i=0;i<n;i++)
            cout<<"第"<<i<<"个数的值"<<dp[i]<<endl;
        return res;
    }
};
  1. 预测赢家

题意：给你一个数组，然后玩家1先拿，玩家2后拿，每次只能拿数组两头的数字，问你玩家1是否一定会胜利。

AC：

class Solution {
public:
    bool PredictTheWinner(vector<int>& nums) {
        //偶数必胜
        //奇数需要讨论
        //每一边拿的策略都是自己拿最大的，留给对手小的。
        //需要考虑我拿两边那个能使的它拿到的最小。
        if(nums.size()/2==0)
        return true;
        int n = nums.size();
        //dp[i][j]代表在i,j范围内，先手比后手拿到最好的多多少。
        vector<vector<int>> dp(n, vector<int>(n, 0));
        //当i==j时，取得的在i-j范围内，最大值即为nums[i]；
        //注意这个下面的初始化顺序，若测试用例为{1,5,2}；
        //是从dp[0][1],dp[1][2],然后顺序是dp[0][2]；
        for (int i = 0; i < n; ++i) dp[i][i] = nums[i];
        
        for (int len = 1; len < n; ++len) {  //定义了步长  
            for (int i = 0, j = len; j < n; ++i, ++j) {
                //当先手拿num[i](左侧),后手拿的最好值为dp[i+1][j]
                //当先手拿num[j](右侧),后手拿的最好值为dp[i][j-1]
                dp[i][j] = max(nums[i] - dp[i + 1][j], nums[j] - dp[i][j - 1]);
                //cout<<nums[i]<<"到"<<nums[j]<<"="<<nums[i]<<"-"<<dp[i + 1][j]<<"和"<<nums[j]<<"-"<<dp[i][j - 1]<<"zuihou:"<<dp[i][j]<<endl;
            }
        }
        //返回dp[0][n-1],因为开始是0，最后一次是n-1,代表在0~n-1范围内先手是否能比后手多
        return dp[0][n-1] >= 0;
    }
};

算法流程描述：

【1，3，6，5，2】 一开始步长给1，就先算dp【0】【1】然后dp【1】【2】、dp【2】【3】、、、

然后步长给2，算 dp【0】【2】 就是【1，3，6】拿走1 -拿走【3，6】的dp，和拿走6 -拿走【1，3】的dp

最后算的是dp【0】【n-1】如果大于0就一定会赢。

121. 买卖股票的最佳时机

示例 1:

输入: [7,1,5,3,6,4] 输出: 5 解释: 在第 2 天（股票价格 = 1）的时候买入，在第 5 天（股票价格 = 6）的时候卖出，最大利润 = 6-1 = 5 。 注意利润不能是 7-1 = 6, 因为卖出价格需要大于买入价格。 示例 2:

输入: [7,6,4,3,1] 输出: 0 解释: 在这种情况下, 没有交易完成, 所以最大利润为 0。

动态规划 前i天的最大收益 = max{前i-1天的最大收益，第i天的价格-前i-1天中的最小价格}

class Solution {
public:
    int maxProfit(vector<int>& prices) {
        int len=prices.size();
        if(len==0)
        return 0;
        vector<int> dp(len+1,0);
        int min=prices[0];
        dp[0]=0;
        for(int i=1;i<len;i++)
        {
            if(prices[i]<min)
            {
                min=prices[i];
            }
            dp[i]=max(dp[i-1],prices[i]-min);
            if(dp[i]<0)
            dp[i]=0;
        }
        return  dp[len-1];
    }
};
  1. 爬楼梯

假设你正在爬楼梯。需要 n 阶你才能到达楼顶。

每次你可以爬 1 或 2 个台阶。你有多少种不同的方法可以爬到楼顶呢？

注意：给定 n 是一个正整数。

示例 1：

输入： 2 输出： 2 解释： 有两种方法可以爬到楼顶。

  1. 1 阶 + 1 阶
  2. 2 阶 示例 2：

输入： 3 输出： 3 解释： 有三种方法可以爬到楼顶。

  1. 1 阶 + 1 阶 + 1 阶
  2. 1 阶 + 2 阶
  3. 2 阶 + 1 阶

这个居然是斐波那契数列：

class Solution {
public:
    int climbStairs(int n) {
        if(n==1||n==2)
        return n;
				//这里如果不判断，会溢出。。。
      	//n==1 后面dp只开了两个 0和1 访问2就越界了。
        vector<int> dp(n+1,0);
        dp[1]=1;
        dp[2]=2;  
        for(int i=3;i<=n;i++)
        {
            dp[i]=dp[i-1]+dp[i-2];
        }
        return dp[n];
    }
};
